              Country country) {
    this.name = name;
    this.country = country;
  }

  public City (City original) {
    this.name = original.name;
    this.country =
          original.country;
  }
}

    public class Country {

  String name;
  City capital;

  public Country(String name,
                 City capital) {
    this.name = name;
    this.capital = capital;

  public Country(Country original){
    this.name = original.name;
    this.capital =

    Static attribute # 

    public class MyClass {

    static int value;
    boolean flag;

    public MyClass(int value, boolean flag){
        this.value = value;
        this.flag = flag;
    }

    void print(){ System.out.print("["+value+" "+flag+"]"); }

    void incrementValue(){ value++; }

    void setFlag(boolean flag){ this.flag = flag; }
}

    MyClass o1 = new MyClass(2, true);
MyClass o2 = new MyClass(3, false);
o1.print();
myMethod(o1);
o1.print();

void myMethod(MyClass object) {
    object.setFlag(false);
    object.incrementValue();
    object.print();
}
